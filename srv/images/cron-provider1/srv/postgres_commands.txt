----- purpleair
CREATE TABLE sensors (
    sensor_index INTEGER PRIMARY KEY,
    name TEXT,
    latitude DOUBLE PRECISION,
    longitude DOUBLE PRECISION
);

CREATE TABLE sensor_readings (
    sensor_index INTEGER REFERENCES sensors(sensor_index),
    batch_time TIMESTAMPTZ NOT NULL,
    last_seen TIMESTAMPTZ NOT NULL,

    humidity SMALLINT,
    pm25_10m REAL,
    pm25_30m REAL,
    pm25_60m REAL,
    pm25_6h REAL,
    pm25_24h REAL,
    PRIMARY KEY (sensor_index, batch_time)
);

----- ab stations
CREATE TABLE stations (
    station_key INTEGER PRIMARY KEY,
    name        TEXT,
    timestamp   TIMESTAMPTZ,
    lat     DOUBLE PRECISION,
    lon     DOUBLE PRECISION,

    AQHI    REAL,
    NO2     REAL,
    O3      REAL,
    PM2     REAL,
    SO2     REAL
);
